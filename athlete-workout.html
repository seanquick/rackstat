import { getAuth, onAuthStateChanged } from "https://www.gstatic.com/firebasejs/10.7.1/firebase-auth.js"; import { getFirestore, doc, getDoc, collection, query, where, getDocs, orderBy } from "https://www.gstatic.com/firebasejs/10.7.1/firebase-firestore.js"; const auth = getAuth(); const db = getFirestore(); // 1. AUTH LISTENER onAuthStateChanged(auth, (user) => { if (user) { window.currentAthleteId = user.uid; loadAvailableWorkouts(); } else { console.log("No athlete logged in."); } }); // 2. HELPER: MATH const calculateWeight = (max, percentage) => { if (!max || !percentage) return "---"; const raw = max * (parseFloat(percentage) / 100); return Math.round(raw / 5) * 5; }; // 3. HELPER: LIFT MAPPING const liftMap = { "bench_press": "benchMax", "back_squat": "squatMax", "power_clean": "cleanMax", "deadlift": "deadliftMax" }; // 4. FUNCTION: LOAD THE LIST OF WORKOUTS window.loadAvailableWorkouts = async () => { const workoutContainer = document.getElementById('available-workouts'); try { const workoutRef = collection(db, "schools", window.currentSchoolId, "workouts"); const q = query(workoutRef, orderBy("createdAt", "desc")); const querySnapshot = await getDocs(q); if (querySnapshot.empty) { workoutContainer.innerHTML = `

No workouts published.

`; return; } workoutContainer.innerHTML = ''; querySnapshot.forEach((doc) => { const data = doc.data(); const btn = document.createElement('button'); btn.className = "w-full bg-gray-800 border border-gray-700 p-5 rounded-xl flex justify-between items-center group text-left mb-3"; btn.onclick = () => loadAthleteWorkout(doc.id, window.currentAthleteId); btn.innerHTML = `

${data.title}

Week ${data.week}
→ `; workoutContainer.appendChild(btn); }); } catch (e) { console.error(e); } }; // 5. FUNCTION: LOAD THE SPECIFIC WORKOUT window.loadAthleteWorkout = async (workoutId, athleteId) => { try { const workoutSnap = await getDoc(doc(db, "schools", window.currentSchoolId, "workouts", workoutId)); const athleteSnap = await getDoc(doc(db, "profiles", athleteId)); if (!workoutSnap.exists()) return; const workout = workoutSnap.data(); const athlete = athleteSnap.exists() ? athleteSnap.data() : {}; // UI Toggle document.getElementById('athlete-dashboard').style.display = 'none'; document.getElementById('athlete-workout-container').style.display = 'block'; document.getElementById('display-title').innerText = workout.title; document.getElementById('display-week').innerText = `WEEK ${workout.week}`; const list = document.getElementById('athlete-exercise-list'); list.innerHTML = ''; workout.exercises.forEach(ex => { const athleteMax = athlete[liftMap[ex.name]] || 0; const card = document.createElement('div'); card.className = "bg-gray-800 border-l-4 border-red-600 p-4 rounded-r-lg mb-4"; let setsHtml = ex.setsData.map((s, i) => { const targetWeight = calculateWeight(athleteMax, s.percent); const setId = `set-${ex.name}-${i}`; // Unique ID for this specific set return `
Set ${i+1} ${targetWeight}LBS
`; }).join(''); card.innerHTML = `

${ex.name.replace(/_/g, ' ')}

${setsHtml}
`; list.appendChild(card); }); window.toggleSetComplete = (setId) => { const container = document.getElementById(`container-${setId}`); const btn = document.getElementById(`btn-${setId}`); // Toggle styles to show completion container.classList.toggle('border-green-500'); container.classList.toggle('bg-green-900/20'); btn.classList.toggle('text-green-500'); }; import { addDoc, serverTimestamp } from "https://www.gstatic.com/firebasejs/10.7.1/firebase-firestore.js"; window.logFullWorkout = async (workoutId) => { try { const loggedExercises = []; const exerciseCards = document.querySelectorAll('.bg-gray-800'); // Your exercise cards exerciseCards.forEach(card => { const name = card.querySelector('h3').innerText; const setRows = card.querySelectorAll('[id^="container-set-"]'); const sets = Array.from(setRows).map(row => { const setId = row.id.replace('container-', ''); return { repsPerformed: document.getElementById(`input-${setId}`).value, completed: row.classList.contains('border-green-500') }; }); loggedExercises.push({ name, sets }); }); // Save to a NEW collection: completed_workouts await addDoc(collection(db, "completed_workouts"), { athleteId: window.currentAthleteId, workoutId: workoutId, schoolId: window.currentSchoolId, timestamp: serverTimestamp(), performance: loggedExercises }); alert("Workout Logged Successfully!"); location.reload(); // Take them back to the dashboard } catch (e) { console.error("Error logging workout:", e); alert("Failed to save workout. Check connection."); } }; } catch (e) { console.error(e); } // Add a big Finish button at the bottom const finishBtn = document.createElement('button'); finishBtn.className = "w-full bg-green-600 hover:bg-green-700 text-white font-black py-4 rounded-xl mt-8 mb-12 shadow-lg tracking-widest uppercase"; finishBtn.innerText = "Log Session & Finish"; finishBtn.onclick = () => window.logFullWorkout(workoutId); list.appendChild(finishBtn); };
